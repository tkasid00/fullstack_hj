*서비스단에서 파일+아이피 처리 방법
```
	@Override
	public int insertForUpload(MultipartFile file, SboardDto dto) {
		if(!file.isEmpty()) {	//파일이 존재하면
			String fileName = file.getOriginalFilename();	//원본 파일 이름 사용
			String uploadPath = "C:/file/";
			File img = new File(uploadPath+fileName);  //임포트는 import java.io.File; 선택
			try {
				
				file.transferTo(img);  //파일 올리기  - 트라이캐치
				dto.setBfile(fileName);
				
			} catch (Exception e) {
				e.printStackTrace();
			}
		}
		try {
			dto.setBip( InetAddress.getLocalHost().getHostAddress()); //아이피 
		} catch (UnknownHostException e) {
			e.printStackTrace();
		}
		return dao.insertForUpload(dto);
	}

```
