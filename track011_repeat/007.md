```java
import java.util.ArrayList;
import java.util.List;
class Solution {
    public String solution(String code) {
        String answer = "";
        int mod=0;
		char[] arr = code.toCharArray();
		List<Character>  ret = new ArrayList(); 
		
		for(int i=0; i<arr.length; i++) {
			if(arr[i]=='1') {
				mod = mod==0? 1:0; continue;
			}			
			switch(mod) {
			case 0 : 
				if(arr[i]!='1' && i%2==0) {
					ret.add(arr[i]);
				} break;
			case 1: 
				if(arr[i]!='1' && i%2!=0) {
					ret.add(arr[i]);
				} break;
			}
		}
        //List는 new String 불가, StringBuilder 처리
		StringBuilder sb = new StringBuilder();
		for (char c : ret) {sb.append(c);}
		answer = sb.length() == 0 ? "EMPTY" : sb.toString();
        return answer;
    }
}
```

```java
class Solution {
    public int solution(int a, int d, boolean[] included) {
        int answer = 0;
        int[] arr = new int[included.length];

		for (int i = 0; i < included.length; i++) {
			if (i == 0) { arr[i] = a;
			} else { arr[i] = arr[i - 1] + d;
			}
			if(included[i]) {answer+=arr[i];}
		}
        return answer;
    }
}
```

```java
class Solution {
    public int solution(int a, int b, int c) {
        int answer = 0;
        int cnt = 0;
		if(a==b) {cnt++;}
		if(a==c) {cnt++;}
		if(c==b) {cnt++;}
		
		switch (cnt) {
		case 0 : answer = a+b+c; break; //다 다른 경우
		case 1 : answer = (a + b + c) * (a*a + b*b + c*c ); break; // 2개 일치
		case 3 : answer =(a + b + c) *(a*a + b*b + c*c ) *(a*a*a + b*b*b + c*c*c ); break; // 3개 일치
		}
        return answer;
    }
}
```

```java
import java.util.ArrayList;
import java.util.List;

class Solution {
    public int solution(int[] num_list) {
        int answer = 0;
        List<Integer> arr1 = new ArrayList<>();
		List<Integer> arr2 = new ArrayList<>();

		for(int i=0; i<num_list.length; i++) {
			if(num_list[i]%2==0) {
				arr2.add(num_list[i]);
			}else {arr1.add(num_list[i]);}
		}
		// 문자열로 변환 후 다시 숫자 변환 
		StringBuilder sb1 = new StringBuilder();
		for(int n :  arr1) {sb1.append(n);}
		StringBuilder sb2 = new StringBuilder();
		for(int n :  arr2) {sb2.append(n);}
		answer = Integer.parseInt(sb1.toString())+Integer.parseInt(sb2.toString());
        return answer;
    }
}
```

```java
class Solution {
	    public int[] solution(int[] num_list) {
	        int[] answer = new int[num_list.length + 1];
	        int length = num_list.length; 
	       // [5, 2, 1, 7, 5] -> [5, 2, 1, 7, 5, 10] 배열 갈이, 리턴값 헷갈림 주의!
	        for (int i=0; i<length; i++) {
	            answer[i] = num_list[i];
	            if (num_list[length - 1] > num_list[length - 2]) {
	                answer[length] = num_list[length - 1] - num_list[length - 2];
	            } else {
	                answer[length] = num_list[length - 1] * 2;
	            }
	        }
	        
	        return answer;
	    }
}
```
