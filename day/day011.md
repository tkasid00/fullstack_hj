    ```bash 
     1) 1~10까지 3의 배수 개수 
     int cnt = 0;
     for(int i=1;i<=10;i++) {if ((i%3) == 0) {cnt++;};}

     2) 소문자 a~z까지 모음의 개수 
     char ch = '\u0000';
	 int cnt = 0;
     for(ch='a'; ch<='z';ch++) {{if (ch=='a'|| ch=='e' || ch=='i'||ch=='o'||ch=='u') {cnt++;} }}
    ```
